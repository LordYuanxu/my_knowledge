# 安装

```bash
# 自定义设置安装位置(默认${HOME}/local/bin下)
export MISE_INSTALL_PATH=$MY_HOME/bin/mise

# 官方安装
curl https://mise.run | sh

# 配置写入~/.bashrc
echo 'eval "$(~/.local/bin/mise activate bash)"' >> ~/.bashrc

# 自定义设置mise软件安装路径
export MISE_DATA_DIR="${MY_HOME}/software"

# 自我升级
mise self-update
```

---
# 使用

查看所有支持的插件

```bash
mise plugins list-all
```

查看可用版本use

```bash
mise ls-remote <plugin-name>
```

安装软件

```bash
mise install <plugin-name>@<version>
```

设置全局或项目级版本

```bash
# 设置全局默认版本
mise use --global <plugin-name>@<version>
# 设置项目级版本
mise use <plugin-name>@<version>
```

可能需要github token(下载请求太多了)

```bash
export GITHUB_TOKEN="ghp_***"
```

```bash
# 安装usage工具
## 使用 mise 安装 usage
mise install usage@latest
## 验证安装
usage --version  # 应显示版本信息

# 生成补全脚本（需要usage）
mise completions bash > ~/sg-users/xuyuan/.bash_completion.mise

# 在 ~/.bashrc 或 ~/.bash_profile 中添加
echo 'source ~/sg-users/xuyuan/.bash_completion.mise' >> ~/sg-users/xuyuan/xy.rc

# 刷新配置
source ~/sg-users/xuyuan/xy.rc
```
