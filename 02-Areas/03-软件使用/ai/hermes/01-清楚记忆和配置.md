# 清除 Hermes Agent 数据目录

```bash
#!/bin/bash
# reset_hermes.sh

echo "停止 Hermes 服务..."
sudo systemctl stop hermes-agent 2>/dev/null || pkill -f hermes
pkill -f "hermes-webui" 2>/dev/null

echo "清除所有数据..."
rm -rf ~/.hermes/
rm -rf /var/lib/hermes/ 2>/dev/null
rm -rf /var/log/hermes/ 2>/dev/null

echo "清除完成，请重新初始化 Hermes"
echo "运行: hermes-agent init 或对应安装命令"
```
