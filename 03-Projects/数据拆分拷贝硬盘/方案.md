
# 识别服务器上插入的设备(`lsblk`命令)
```bash
lsblk --list --json -o PATH,FSSIZE,FSAVAIL,FSUSED,FSUSE%,FSTYPE,LABEL,UUID,PTUUID,PARTUUID,WWN,MOUNTPOINT \
| jq --argjson labels '["mj22","mj23", "mj24"]' '.blockdevices[] | select(.label | IN($labels[])) | {(.label): .}' \
| jq -s add
