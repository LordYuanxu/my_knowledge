```
#!/bin/bash
#SBATCH --job-name=writeFasta
#SBATCH --partition=chaifen
#SBATCH --nodes=1
#SBATCH --ntasks=1
#SBATCH --cpus-per-task=18
#SBATCH --mem=128G
#SBATCH --output=%x_%j.out
#SBATCH --error=%x_%j.err
#SBATCH --time=24:00:00

SIF_IMAGE="/mnt/dlustre/users/sanger/sg-users/xuyuan/majorbio_task/datasplit_v3/data_split/library_split/software_test/writeFaTool_v55.13_Linux_SURFSeqQ/writeFasta.sif"
EXECUTABLE="/opt/writeFasta/writeFasta_v55.13"
CONFIG="//mnt/dlustre/users/sanger/sg-users/xuyuan/majorbio_task/datasplit_v3/data_split/library_split/software_test/writeFaTool_v55.13_Linux_SURFSeqQ/wfq.config"

# time 日志单独存放，-a 表示追加
TIME_LOG="time_${SLURM_JOB_ID}.log"

echo "=== Job Start: $(date '+%Y-%m-%d %H:%M:%S') | JobID=$SLURM_JOB_ID ===" > "$TIME_LOG"

# 用 -o 指定 time 输出到文件，命令的 stderr 仍走 sbatch 的 %x_%j.err
/usr/bin/time -v -a -o "$TIME_LOG" \
    singularity exec \
        --bind /mnt/dlustre:/mnt/dlustre \
        "$SIF_IMAGE" \
        "$EXECUTABLE" \
        "$CONFIG"

EXIT_CODE=$?
echo "=== Job End: $(date '+%Y-%m-%d %H:%M:%S') | Exit=$EXIT_CODE ===" >> "$TIME_LOG"

# 追加 SLURM 统计
echo "" >> "$TIME_LOG"
/usr/bin/sacct -j $SLURM_JOB_ID --format=JobID,Elapsed,CPUTime,TotalCPU,MaxRSS,ReqMem,AllocCPUS >> "$TIME_LOG" 2>/dev/null

exit $EXIT_CODE
```

1. 如果重运行，会覆盖原来的结果，所以可以直接重新运行不需要删除原来的结果
2. 目前虽然可以不同Lane同时拆分，但是软件内部无法并行
